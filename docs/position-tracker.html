// Array to store your positions dynamically let positions = []; // Configuration const UPDATE_INTERVAL_MS = 15000; // Fetch new prices every 15 seconds const CORS_PROXY = "https://corsproxy.io/?"; // Free proxy to bypass CORS // --- Core Math Functions --- function calculatePnL(entry, current) { if (!current) return 0; return (((current - entry) / entry) * 100).toFixed(2); } function calculateRMultiple(entry, sl, current) { if (!current) return 0; const risk = entry - sl; if (risk <= 0) return 0; // Prevent division by zero const reward = current - entry; return (reward / risk).toFixed(2); } // --- Fetch Live Prices --- async function fetchLivePrices() { if (positions.length === 0) return; // Skip fetching if there are no positions for (let i = 0; i < positions.length; i++) { const pos = positions[i]; try { // Yahoo Finance API endpoint for current pricing const url = `https://query1.finance.yahoo.com/v8/finance/chart/${pos.symbol}?interval=1m`; const response = await fetch(CORS_PROXY + encodeURIComponent(url)); const data = await response.json(); // Extract the current market price const livePrice = data.chart.result[0].meta.regularMarketPrice; if (livePrice) { positions[i].currentPrice = livePrice; } } catch (error) { console.error(`Failed to fetch price for ${pos.symbol}:`, error); } } // Re-render table and summary after all prices are updated renderTable(); updateSummary(); } // --- Rendering Functions --- function renderTable() { const tbody = document.getElementById("positionsTable"); // SAFETY CHECK: If we aren't on the tracker page, stop running. if (!tbody) return; tbody.innerHTML = ""; if (positions.length === 0) { // RESTORED HTML TAGS tbody.innerHTML = `No open positions found.`; return; } positions.forEach(pos => { const currentP = pos.currentPrice || pos.entry; const pnl = calculatePnL(pos.entry, currentP); const rMult = calculateRMultiple(pos.entry, pos.sl, currentP); const pnlColor = pnl >= 0 ? "color: #00c853;" : "color: #ff3d00;"; const rColor = rMult >= 0 ? "color: #00c853;" : "color: #ff3d00;"; const tr = document.createElement("tr"); // RESTORED HTML TAGS tr.innerHTML = ` ${pos.symbol} ${pos.owner} ₹${parseFloat(pos.entry).toFixed(2)} ₹${parseFloat(pos.sl).toFixed(2)} ${pos.qty} ₹${parseFloat(currentP).toFixed(2)} ${pnl}% ${rMult}R `; tbody.appendChild(tr); }); } function updateSummary() { // SAFETY CHECK: Ensure summary elements exist before updating const openCountEl = document.getElementById("openCount"); if (!openCountEl) return; const openCount = positions.length; let totalR = 0; let winners = 0; let losers = 0; positions.forEach(pos => { const currentP = pos.currentPrice || pos.entry; const rMult = parseFloat(calculateRMultiple(pos.entry, pos.sl, currentP)); totalR += rMult; if (rMult > 0) winners++; else if (rMult < 0) losers++; }); const avgR = openCount > 0 ? (totalR / openCount).toFixed(2) : 0; openCountEl.innerText = openCount; document.getElementById("avgR").innerText = `${avgR}R`; document.getElementById("winLoss").innerText = `${winners} / ${losers}`; document.getElementById("heat").innerText = openCount > 0 ? "Active" : "Neutral"; } // --- Actions --- window.deletePosition = function(id) { positions = positions.filter(p => p.id !== id); renderTable(); updateSummary(); }; // --- Initialization --- document.addEventListener("DOMContentLoaded", () => { // Initial Render renderTable(); updateSummary(); // Start real-time price loop fetchLivePrices(); setInterval(fetchLivePrices, UPDATE_INTERVAL_MS); });
